47 void IPTVTopologyOracle::generateUserViewMap(Scheduler* scheduler) {
48  BOOST_LOG_TRIVIAL(debug) << "entering generateUserViewMap";
49  /* fetching video chunks takes time, so we get delays with respect to the
50  * session defined here; hence I'm introducing a guardThreshold so that we
51  * have enough time to complete all the flows before midnight (hopefully)
52 
53  uint guardThreshold = 60;
54  uint roundDuration = scheduler->getRoundDuration() - guardThreshold;
55  * */
56  uint roundDuration = scheduler->getRoundDuration();
57  double totalHours = 0, randomHours;
58  boost::random::normal_distribution<> userSessionDist(avgHoursPerUser);
59  boost::random::discrete_distribution<> hourDist(usrPctgByHour);
60  boost::random::uniform_int_distribution<> minSecDist(0, 3599);
61  // Generate a session start time and session length for each user
62  SimTime sessionStart, sessionLength;
63 
64  BOOST_FOREACH(Vertex v, topo->getPonNodes()) {
65  for (uint i = 0; i < topo->getPonCustomers(v); i++) {
66  PonUser user = std::make_pair(v,i);
67  // populate userWatchMap for the first time if it's round 0
68  if (scheduler->getCurrentRound() == 0) {
69  std::pair<PonUser, UserWatchingInfo> userWatchMapEntry = std::make_pair(user, UserWatchingInfo());
70  userWatchMap.insert(userWatchMapEntry);
71  }
72  else {
73  userWatchMap.at(user).reset();
74  }
75  randomHours = userSessionDist(gen);
76  // daily sessions can't be longer than a day, duh
77  if (randomHours > 24)
78  randomHours = 24;
79  if (randomHours > 0)
80  totalHours += randomHours;
81  sessionLength = std::floor((randomHours * 3600) + 0.5);
82  if (sessionLength <= 0) {
83  BOOST_LOG_TRIVIAL(info) << "TopologyOracle::generateUserViewMap() - "
84  "non-positive sessionLength for user " << v << "," << i
85  << " - skipping it";
86  continue;
87  }
88  /* the time we generate is assumed to be the middle point of the viewing
89  * session, so we shift it of half sessionLength, but we need to ensure it
90  * still fits inside the round
91  */
92  sessionStart = (hourDist(gen) * 3600 + minSecDist(gen)) -
93  std::floor((sessionLength / 2) + 0.5);
94  if (sessionStart < 0)
95  sessionStart = 0;
96  else if (sessionStart + sessionLength >= roundDuration)
97  sessionStart = roundDuration - sessionLength;
98  SimTimeInterval interval(sessionStart, sessionStart + sessionLength);
99  // store the new interval
100  userWatchMap.at(user).dailySessionInterval = interval;
101  // generate first request and schedule it
102  this->generateNewRequest(user, interval.getStart(), scheduler);
103  }
104  }
105  BOOST_LOG_TRIVIAL(info) << "Total hours of scheduled viewing for the current day: "
106  << totalHours << "; avg. hours per user: "
107  << totalHours/(topo->getNumCustomers())
108  << " (requested: " << avgHoursPerUser << ")";
109 }

161  * determines how long the user is going to be watching it (because of zapping)
162  * and stores the required information in the userWatchMap, then creates a new
163  * Flow and schedules it.

165 void IPTVTopologyOracle::generateNewRequest(PonUser user, SimTime time,
166  Scheduler* scheduler) {
167  SimTime sessionEnd = userWatchMap.at(user).dailySessionInterval.getEnd();
168  if (time < sessionEnd) {
169  boost::random::uniform_int_distribution<> indexDist(0, 17);
170  uint i = (*relDayDist)(gen);
171  ContentElement* content = dailyCatalog[i].at((*rankDist)(gen));
172  double watchingPortion = sessionLength[indexDist(gen)];
173  BOOST_LOG_TRIVIAL(debug) << "At time " << time
174  << " generated request from user " << user.first
175  << "," << user.second << " for content " << content->getName()
176  << ", watchingPortion = " << watchingPortion;
177  /*
178  Capacity reqLength = watchingPortion * content->getSize();
179  // calculate the time at which the user will be done with this content
180  SimTime endTime = time + std::ceil(reqLength / this->bitrate);
181  // check that the new request would not go past the desired session
182  // length; this will make all requests past midnight end at midnight :(
183  if (endTime > sessionEnd) {
184  endTime = sessionEnd;
185  }
186  */
187  // determine the number of chunks to be watched for this content
188  // TODO: reduce the number of chunks to be watched if we're going past the sessionEnd deadline
189  uint watchingChunks = std::ceil(watchingPortion * content->getTotalChunks());
190  if (watchingChunks > content->getTotalChunks())
191  watchingChunks = content->getTotalChunks();
192  UserWatchingMap::iterator wIt = userWatchMap.find(user);
193  assert (wIt != userWatchMap.end());
194  wIt->second.content = content;
195  wIt->second.chunksToBeWatched = watchingChunks;
196  wIt->second.currentChunk = 0;
197  // we are waiting for the first chunk to be downloaded
198  wIt->second.waiting = true;
199  /* Request enough chunks to fill the buffer.
200  */
201  for (uint i = 0; i < bufferSize; i++) {
202  // stop if we fetched all the chunks of this content
203  if (i >= content->getTotalChunks())
204  break;
205  else {
206  Flow* request = new Flow(content, user, time, i);
207  scheduler->schedule(request);
208  wIt->second.highestChunkFetched = i;
209  BOOST_LOG_TRIVIAL(debug) << "Fetching chunk " << i << " of content "
210  << content->getName() << ", highestChunkFetched = "
211  << userWatchMap.at(user).highestChunkFetched;
212  }
213  }
214  }

225 void IPTVTopologyOracle::preCache() {
226  bool isFull = false;
227  uint i(0), day(0), index(0);
228  ContentElement* content = nullptr;
229  while (!isFull) {
230  day = std::floor(i / dailyCatalog.at(0).size()); // all the same size
231  if (day >= dailyCatalog.size())
232  break;
233  index = i % dailyCatalog.at(day).size();
234  content = dailyCatalog.at(day).at(index);
235  for (uint as = 0; as < topo->getNumASes(); as++) {
236  if (localCacheMap->at(as).fitsInCache(content->getSize())) {
237  auto chunks = content->getChunks();
238  std::pair<bool, std::set<ChunkPtr> > result;
239  for (auto j = chunks.begin(); j != chunks.end(); j++) {
240  result = localCacheMap->at(as).addToCache(*j, (*j)->getSize(), 0);
241  assert(result.first == true);
242  assert(result.second.empty() == true);
243  assert(localCacheMap->at(as).isCached(*j));
244  }
245  }
246  else {
247  isFull = true;
248  break;
249  }
250  }
251  i++;
252  }
253  BOOST_LOG_TRIVIAL(trace) << "cached " << i << " contents in AS caches";
254 }
