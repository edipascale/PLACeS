103 void TopologyOracle::addToCache(PonUser user, ChunkPtr chunk, SimTime time) {
104  if (!chunk) {
105  BOOST_LOG_TRIVIAL(error) << "TopologyOracle::addToCache() - invalid ChunkPtr";
106  abort();
107  }
108  ContentElement* content = chunk->getContent();
109  uint chunkIndex = chunk->getIndex();
110  // update the chunkMap for content retrieval
111  uint asid = topo->getAsid(user);
112  ChunkMap::iterator cIt = asidContentMap->at(asid).find(chunk);
113  std::pair<bool, std::set<ChunkPtr> > addResult;
114  if (cIt == asidContentMap->at(asid).end()) {
115  // This is the first time the oracle sees this chunk, add it to the map
116  // Note: this should not happen now, throw an error
117  BOOST_LOG_TRIVIAL(error) << time << ":ERROR: TopologyOracle::addToCache() - "
118  "unknown chunk for content " << content->getName() <<
119  ", chunkId " << chunkIndex;
120  abort();
121  // std::set<PonUser> sourceNodes;
122  // cIt = asidContentMap->at(asid).insert(std::make_pair(content, sourceNodes)).first;
123  }
124  // update userCacheMap according to the specified cache policy
125  BOOST_LOG_TRIVIAL(trace) << time << ": caching chunk " <<
126  chunk->getIndex() << " of content " << content->getName() <<
127  " at User " << user.first << "," << user.second;
128  addResult = userCacheMap->at(user).addToCache(chunk, chunk->getSize(), time);
129  if (!addResult.first) {
130  if (userCacheMap->at(user).getMaxSize() >= chunk->getSize()) {
131  BOOST_LOG_TRIVIAL(warning) << time << ": WARNING: failed to cache chunk "
132  << chunk->getIndex() << " of content "
133  << content->getName()
134  << " at User " << user.first << "," << user.second;
135  }
136  } else {
137  /* if caching at the user was successful, add the user to the asidCacheMap
138  * for its AS (cIt is the iterator to the right position in the map, see above)
139  * note that the user might be already there if he had a copy of the content
140  * with a different size
141  */
142  if (cIt->second.find(user) == cIt->second.end()) {
143  bool insert = cIt->second.insert(user).second;
144  if (!insert) {
145  BOOST_LOG_TRIVIAL(warning) << time << ": WARNING: failed to insert User "
146  << user.first << "," << user.second << " as a source for chunk "
147  << chunk->getIndex() << " of content "
148  << content->getName() << " in AS " << asid;
149  }
150  }
151  }
152  BOOST_FOREACH (ChunkPtr e, addResult.second) {
153  // erase the content from the ContentMap entry of the user
154  this->removeFromCMap(e, user);
155  // update the asidContentMap too
156  cIt = asidContentMap->at(asid).find(e);
157  if (cIt != asidContentMap->at(asid).end())
158  cIt->second.erase(user);
159  }
160 
161  if (!reducedCaching && !preCaching) {
162  // also update the local Cache associated to this user ASID
163  // (the flow is not simulated as it is assumed it was cached as it transited
164  // towards the user)
165  Vertex lCache = topo->getLocalCache(user.first);
166  if (!localCacheMap->at(lCache).isCached(chunk)) {
167  // debug info
168  BOOST_LOG_TRIVIAL(trace) << time << ": caching chunk " <<
169  chunk->getIndex() << " of content " << content->getName() <<
170  " at local cache " << lCache;
171  addResult = localCacheMap->at(lCache).addToCache(chunk, chunk->getSize(), time);
172  if (!addResult.first) {
173  BOOST_LOG_TRIVIAL(warning) << time << ": WARNING: failed to cache caching chunk " <<
174  chunk->getIndex() << " of content " << content->getName()
175  << " at AS cache " << lCache;
176  }
177  }
178  }

207 bool TopologyOracle::serveRequest(Flow* flow, Scheduler* scheduler) {
208  PonUser destination = flow->getDestination();
209  PonUser closestSource = UNKNOWN;
210  SimTime time = scheduler->getSimTime();
211  int currentDay = scheduler->getCurrentRound();
212  ContentElement* content = flow->getContent();
213  std::string contentName = content->getName();
214  uint chunkId = flow->getChunkId();
215  const ChunkPtr chunk = content->getChunkById(chunkId);
216  BOOST_LOG_TRIVIAL(trace) << time << ": fetching source for chunk " << chunkId
217  << " of content " << contentName
218  << " to user " << destination.first << "," << destination.second;
219  // update the number of requests for this content if it's the first chunk
220  if (chunk->getIndex() == 0)
221  dailyRanking.at(currentDay-content->getReleaseDay()).hit(content);
222  // also increase the number of hits of the chunk (currently not used)
223  // chunk->increaseViewsThisRound();
224  // Set the flow as a transfer, since we are now assigning the source
225  flow->setFlowType(FlowType::TRANSFER);
226  // First check if the content is already in the user cache: if it is, there's
227  // no need to simulate the data exchange and we only need to update the
228  // cache stats.
229  if (checkIfCached(destination, chunk)) {
230  // signal to the scheduler that this flow is "virtual"
231  flow->setSource(destination);
232  flow->setEta(time);
233  // update this user's cacheMap entry (for LRU/LFU)
234  bool result = userCacheMap->at(destination).getFromCache(chunk,
235  (scheduler->getCurrentRound()+1)*time, true);
236  assert(result);
237  flowStats.servedRequests.at(scheduler->getCurrentRound())++;
238  flowStats.completedRequests.at(scheduler->getCurrentRound())++;
239  flowStats.localRequests.at(scheduler->getCurrentRound())++;
240  flowStats.fromPeers.at(scheduler->getCurrentRound())++;
241  // debug info
242  BOOST_LOG_TRIVIAL(debug) << time << ": user " << destination.first
243  << "," << destination.second
244  << " had a local copy of chunk " << chunkId << " from content "
245  << contentName;
246  return true;
247  }
248  // try to find a local peer first
249  uint asid = topo->getAsid(destination);
250  std::set<PonUser> localSources = asidContentMap->at(asid).at(chunk);
251  // copy the set to a vector which can then be randomized to reduce the chance
252  // the same peer will be selected over and over until congestion is reached
253  std::vector<PonUser> randSources(localSources.begin(), localSources.end());
254  std::random_shuffle(randSources.begin(), randSources.end());
255  localSources.clear();
256  bool foundSource = false;
257  // Attempts to find the closest source whose route to the destination is not congested
258  while (!randSources.empty() && !foundSource) {
259  closestSource = randSources.back();
260  // check if the identified source has enough content to serve this request
261  if (this->checkIfCached(closestSource, chunk)
262  && !topo->isCongested(closestSource, destination))
263  foundSource = true;
264  else {
265  // erase closestSource from randSources
266  randSources.pop_back();
267  closestSource = UNKNOWN;
268  }
269  }
270  if (!foundSource)
271  {
272  // Couldn't find a local source for the required content, check for the
273  // local AS cache (unless we're in reducedCaching mode)
274 
275  if (!reducedCaching)
276  {
277  Vertex lCache = topo->getLocalCache(flow->getDestination().first);
278  if (checkIfCached(lCache, chunk)) {
279  // NOTE: We should check here if the downlink is congested, but with our
280  // current hypothesis it should never be the case
281  assert(topo->isCongested(std::make_pair(lCache, 0), destination) == false);
282  // debug info
283  BOOST_LOG_TRIVIAL(trace) << time << ": user " << destination.first
284  << "," << destination.second
285  << " downloading chunk " << chunkId
286  << " of content " << contentName
287  << " from AS cache node " << lCache
288  << " (closestSource=" << closestSource.first
289  << "," << closestSource.second << ")";
290  this->getFromLocalCache(lCache, chunk,
291  (scheduler->getCurrentRound()+1)*time);
292  flow->setSource(std::make_pair(lCache, 0));
293  flowStats.servedRequests.at(scheduler->getCurrentRound())++;
294  flowStats.localRequests.at(scheduler->getCurrentRound())++;
295  flow->setP2PFlow(false);
296  scheduler->schedule(flow);
297  topo->updateCapacity(flow, scheduler, true);
298  return true;
299  }
300  }
301  // no local peer and no micro cache, look for non local peer
302  BOOST_LOG_TRIVIAL(trace) << "No local source found for chunk " << chunkId
303  << " of content "+
304  contentName+" , searching for non-local P2P sources";
305  uint minDistance, distance, asIndex;
306  ChunkMap::iterator cIt;
307  // vector of ASes which have already explored for viable sources
308  std::vector<bool> exploredASes(topo->getNumASes(), false);
309  exploredASes[asid] = true;
310  foundSource = false;
311  while (std::find(exploredASes.begin(), exploredASes.end(), false) != exploredASes.end()
312  && !foundSource) {
313  // identify an AS with minimum distance from destination
314  minDistance = INF_TIME;
315  for (uint i = 0; i < topo->getNumASes(); i++) {
316  if (exploredASes[i])
317  continue;
318  cIt = asidContentMap->at(i).find(chunk);
319  if (cIt == asidContentMap->at(i).end() || cIt->second.empty()) {
320  // this AS has no source with the required chunk, mark it as explored
321  // to save time at a future iteration
322  BOOST_LOG_TRIVIAL(trace) << "No viable source for chunk " << chunkId
323  << " of content " + contentName + " in asid " << i;
324  exploredASes[i] = true;
325  } else {
326  distance = topo->getRoute(*(cIt->second.begin()), destination).size();
327  if (distance < minDistance) {
328  minDistance = distance;
329  asIndex = i;
330  }
331  }
332  }
333  // scan available sources in the identified closer AS
334  if (minDistance != INF_TIME) {
335  // Attempts to find the closest source whose route to the destination is not congested
336  localSources.clear();
337  localSources = asidContentMap->at(asIndex).at(chunk);
338  randSources.clear();
339  randSources.assign(localSources.begin(), localSources.end());
340  std::random_shuffle(randSources.begin(), randSources.end());
341  localSources.clear();
342  while (!randSources.empty() && !foundSource) {
343  closestSource = randSources.back();
344  // check if the identified source has enough content to serve this request
345  if (this->checkIfCached(closestSource, chunk)
346  && !topo->isCongested(closestSource, destination))
347  foundSource = true;
348  else {
349  // erase closestSource from randSources
350  randSources.pop_back();
351  closestSource = UNKNOWN;
352  }
353  }
354  exploredASes[asIndex] = true;
355  }
356  }
357  }
358  // if the closestSource is still unknown, we have to go to the central server
359  if (closestSource == UNKNOWN) {
360  // Ensure the central server was specified in the topology file
361  Vertex centralServer = topo->getCentralServer();
362  assert(centralServer != std::numeric_limits<unsigned int>::max());
363  // The 1 in the second PonUser field is to differentiate it from an AS Cache
364  closestSource = std::make_pair(centralServer, 1);
365  if (topo->isCongested(closestSource, destination)) {
366  // This was our last hope: there is no uncongested route to any source
367  flowStats.congestionBlocked.at(scheduler->getCurrentRound())++;
368  BOOST_LOG_TRIVIAL(info) << time << ": user " << destination.first
369  << "," << destination.second
370  << " could not find an un-congested route to chunk " << chunkId
371  << " of content " << contentName;
372  return false;
373  }
374  // If we're in reducedCaching mode, check if the content is stored on the
375  // central cache, otherwise fetch it off-network
376  if (reducedCaching) {
377  if (!checkIfCached(centralServer, chunk)) {
378  localCacheMap->at(centralServer).addToCache(chunk,
379  chunk->getSize(),
380  (scheduler->getCurrentRound()*roundDuration)+time);
381  }
382  // update LFU/LRU stats
383  this->getFromLocalCache(centralServer, chunk,
384  (scheduler->getCurrentRound()+1)*time);
385  }
386  flow->setP2PFlow(false);
387  // debug info
388  BOOST_LOG_TRIVIAL(trace) << time << ": user " << destination.first
389  << "," << destination.second
390  << " downloading chunk " << chunkId
391  << " of content " << contentName
392  << " from central server";
393  } else {
394  // p2p flow, update user cache statistics (for LFU/LRU purposes)
395  flow->setP2PFlow(true);
396  bool result = userCacheMap->at(closestSource).getFromCache(chunk,
397  (scheduler->getCurrentRound()+1)*time, false);
398  assert(result);
399  // check for locality is done here to avoid central server to be mistakenly
400  // identified as local
401  if (topo->isLocal(destination.first, closestSource.first))
402  flowStats.localRequests.at(scheduler->getCurrentRound())++;
403  // debug info
404  BOOST_LOG_TRIVIAL(trace) << time << ": user " << destination.first
405  << "," << destination.second << " (asid " << topo->getAsid(destination)
406  << ") downloading chunk " << chunkId
407  << " of content " << contentName
408  << " from peer " << closestSource.first << ","
409  << closestSource.second << " (asid " << topo->getAsid(destination)
410  << ")";
411  }
412  // Assign the closest source to the flow
413  flow->setSource(closestSource);
414  // Update flow statistics
415  flowStats.servedRequests.at(scheduler->getCurrentRound())++;
416  // schedule flow with INF_TIME ETA so that it's in the queue for future updating
417  scheduler->schedule(flow);
418  // update network load and estimate ETA based on available bandwidth
419  topo->updateCapacity(flow, scheduler, true);
420  return true;
421 }

423 void TopologyOracle::notifyCompletedFlow(Flow* flow, Scheduler* scheduler) {
424  PonUser dest = flow->getDestination();
425  SimTime time = scheduler->getSimTime();
426  uint round = scheduler->getCurrentRound();
427  const ChunkPtr chunk = flow->getContent()->getChunkById(flow->getChunkId());
428  switch(flow->getFlowType()) {
429  case FlowType::TRANSFER:
430  {
431  if (flow->getSource() != dest) {
432  BOOST_LOG_TRIVIAL(debug) << "At time " << time << " user " << dest.first
433  << "," << dest.second << " completed download of chunk "
434  << chunk->getIndex() << " of content " << chunk->getContent()->getName();
435  flow->updateSizeDownloaded(time);
436  if (flow->getSizeDownloaded() < chunkSize) {
437  // ensure that this is a result of a discrete time scale (approximation to previous second)
438  assert(chunk->getSize() - flow->getSizeDownloaded() <= flow->getBandwidth());
439  BOOST_LOG_TRIVIAL(trace) << time << ": completed flow has sizeDownloaded (" <<
440  flow->getSizeDownloaded() << ") < Chunk Size (" <<
441  chunkSize << ") due to time approximation, fixing this";
442  flow->setSizeDownloaded(chunkSize);
443  }
444  // update flow statistics
445  SimTime flowDuration = time - flow->getStart();
446  // Update average flow duration stats, both generic and p2p / cache ones
447  flowStats.avgFlowDuration.at(round) = (flowStats.avgFlowDuration.at(round)
448  * flowStats.completedRequests.at(round)
449  + flowDuration) / (flowStats.completedRequests.at(round) + 1);
450  flowStats.completedRequests.at(round)++;
451  BOOST_LOG_TRIVIAL(trace) << "Flow duration: " << flowDuration <<
452  "; avgFlowDuration: " << flowStats.avgFlowDuration.at(round);
453 
454  if (flow->isP2PFlow()) {
455  flowStats.avgPeerFlowDuration.at(round) = (flowStats.avgPeerFlowDuration.at(round)
456  * flowStats.fromPeers.at(round)
457  + flowDuration) / (flowStats.fromPeers.at(round) + 1);
458  flowStats.fromPeers.at(round)++;
459  } else {
460  flowStats.avgCacheFlowDuration.at(round) = (flowStats.avgCacheFlowDuration.at(round) *
461  (flowStats.fromASCache.at(round) + flowStats.fromCentralServer.at(round)) + flowDuration)
462  / (flowStats.fromASCache.at(round) + flowStats.fromCentralServer.at(round) + 1);
463  if (flow->getSource().first == topo->getCentralServer() &&
464  flow->getSource().second == 1)
465  flowStats.fromCentralServer.at(round)++;
466  else
467  flowStats.fromASCache.at(round)++;
468  }
469  topo->updateLoadMap(flow);
470  // notify the source cache that it has completed this upload
471  PonUser source = flow->getSource();
472  bool retValue(false);
473  if (flow->isP2PFlow())
474  retValue = userCacheMap->at(source).uploadCompleted(chunk);
475  else
476  retValue = localCacheMap->at(source.first).uploadCompleted(chunk);
477  assert(retValue);
478  // update cache info (unless the content has expired, e.g. a flow carried over
479  // from the previous round)
480  /* FIXME: checking for a nullptr here does not make any sense, as we have
481  already accessed it earlier in the method - check the code to see if we need
482  to do it earlier or it's not needed at all*/
483  if (flow->getContent() != nullptr) {
484  /* here we need to ask the oracle to decide whether we should cache the new
485  * chunk. Because the optimization will also tell us what to delete, we
486  * have to remove those chunks manually before invoking addToCache
487  * Note that in the popularity estimation branch, we should try to optimize
488  * only when we know enough about the content - e.g. a round.
489  */
490  if (round == 0 || !cachingOpt) {
491  this->addToCache(dest, chunk, round * roundDuration + time);
492  } else {
493  std::pair<bool, bool> optResult = this->optimizeCaching(dest, chunk,
494  time, round);
495  /* we need to add the requested element if the optimization failed OR if
496  * it succeeded and it determined that we need to. The difference is that
497  * in the second case elements that need to be removed have already been erased
498  */
499  if (!optResult.first || (optResult.first && optResult.second)) {
500  this->addToCache(dest, chunk, round * roundDuration + time);
501  }
502  // also record if the cache optimization was successful
503  if (optResult.first == true)
504  flowStats.cacheOptimized.at(round)++;
505  }
506  }
507  // free resources in the topology
508  topo->updateCapacity(flow, scheduler, false);
509  }
510  // if the flow is carried over from the previous round, the rest should be skipped
511  if (flow->getContent() == userWatchMap.at(dest).content) {
512  // add this chunk to the streaming buffer of the user
513  // FIXME: we are not checking if the buffer is full, although we do check when we fetch
514  auto insResult = userWatchMap.at(dest).buffer.insert(chunk);
515  assert(insResult.second == true);
516  /* here we previously generated a new request, however this is no longer related to
517  * downloads, but rather to watching. the only thing we need to do is starting
518  * a watch event if the user was waiting for the chunk we just downloaded
519  */
520  if (userWatchMap.at(dest).waiting
521  && userWatchMap.at(dest).currentChunk == chunk->getIndex()) {
522  // if we are waiting for a chunk we can't be done, just make sure
523  assert(chunk->getIndex() < userWatchMap.at(dest).chunksToBeWatched);
524  // start a new watching flow for the chunk we just got
525  BOOST_LOG_TRIVIAL(debug) << "User was waiting for this chunk, starting a WATCH flow";
526  // if it was not the first chunk, display a debugging message to track rebuffering events
527  if (chunk->getIndex() > 0) {
528  BOOST_LOG_TRIVIAL(info) << "At time " << scheduler->getSimTime()
529  << " user " << dest.first << "," << dest.second
530  << " stopped waiting for chunk " << chunk->getIndex() << " of content "
531  << chunk->getContent()->getName();
532  }
533  SimTime eta = scheduler->getSimTime() +
534  std::ceil(chunk->getSize() / this->bitrate);
535  Flow* watchEvent = new Flow(flow->getContent(), dest, eta, chunk->getIndex(),
536  FlowType::WATCH);
537  scheduler->schedule(watchEvent);
538  userWatchMap.at(dest).waiting = false;
539  }
540  } else {
541  BOOST_LOG_TRIVIAL(debug) << "Carried over transfer of chunk "
542  << chunk->getIndex() << " of content " << chunk->getContent()->getName()
543  << " to user " << dest.first << "," << dest.second
544  << " from previous round";
545  }
546  }
547  break;
548 
549  case FlowType::WATCH:
550  {
551  BOOST_LOG_TRIVIAL(debug) << "At time " << time << " user " << dest.first
552  << "," << dest.second << " finished watching chunk "
553  << chunk->getIndex() << " of content " << chunk->getContent()->getName();
554  // is it a carried over flow?
555  if (flow->getContent() != userWatchMap.at(dest).content) {
556  /* the userWatchMap for this watching session has been overwritten now,
557  * and a new session has been programmed, so there is no reason to
558  * keep going with this but to complete the flows that were still hanging
559  */
560  BOOST_LOG_TRIVIAL(info) << "carried over watch flow for content " <<
561  flow->getContent()->getName() << ", currently watching " <<
562  userWatchMap.at(dest).content->getName();
563  return;
564  }
565  // are we done with this content?
566  uint completedChunk = userWatchMap.at(dest).currentChunk;
567  if (time >= userWatchMap.at(dest).dailySessionInterval.getEnd() ||
568  completedChunk >= userWatchMap.at(dest).chunksToBeWatched-1) {
569  // yes, we are; free buffer and reset watching info
570  BOOST_LOG_TRIVIAL(debug) << "At time " << time << " user " << dest.first
571  << "," << dest.second << " finished watching content "
572  << flow->getContent()->getName() << " with chunk "
573  << completedChunk << "/" << flow->getContent()->getTotalChunks()-1;
574  userWatchMap.at(dest).reset();
575  // generate a new request (if the daily session is over, it's going to be checked there)
576  this->generateNewRequest(dest, time, scheduler);
577  // and that's all
578  return;
579  }
580  // still more watching to do for this content
581  // free space in the buffer now that the chunk has been watched
582  userWatchMap.at(dest).buffer.erase(flow->getContent()->getChunkById(completedChunk));
583  // pre-fetch as many new chunks as we can, since there's space in the buffer now
584  uint bufferSlots = this->bufferSize - userWatchMap.at(dest).buffer.size();
585  // there should be at least one free slot if nothing is wrong!
586  assert (bufferSlots > 0);
587  while (bufferSlots > 0 &&
588  userWatchMap.at(dest).highestChunkFetched < flow->getContent()->getTotalChunks()-1) {
589  BOOST_LOG_TRIVIAL(debug) << "There's " << bufferSlots << " slots in the buffer, "
590  "pre-fetching chunk " << userWatchMap.at(dest).highestChunkFetched+1;
591  Flow* requestChunk = new Flow(flow->getContent(), dest, time,
592  userWatchMap.at(dest).highestChunkFetched+1);
593  scheduler->schedule(requestChunk);
594  userWatchMap.at(dest).highestChunkFetched++;
595  bufferSlots--;
596  }
597 
598  // first make sure that we are not attempting to fetch a chunk that does not exist
599  assert(completedChunk != flow->getContent()->getTotalChunks()-1);
600  // update the chunk we are currently interested into
601  userWatchMap.at(dest).currentChunk++;
602  // check if we've got the next chunk so we can start watching straight away
603  if (userWatchMap.at(dest).buffer.find(flow->getContent()->getChunkById(completedChunk+1))
604  != userWatchMap.at(dest).buffer.end()) {
605  // start a new watching flow
606  BOOST_LOG_TRIVIAL(debug) << "We had the next chunk (" << completedChunk+1
607  << ") in the buffer, starting a new WATCH flow";
608  SimTime eta = scheduler->getSimTime() +
609  std::ceil(flow->getContent()->getChunkById(completedChunk+1)->getSize() / this->bitrate);
610  Flow* watchEvent = new Flow(flow->getContent(), dest, eta, completedChunk+1,
611  FlowType::WATCH);
612  scheduler->schedule(watchEvent);
613  // and remove the waiting status in case it was set
614  userWatchMap.at(dest).waiting = false;
615  } else {
616  // there are chunks we want to watch and we do not have them. ALARM!
617  BOOST_LOG_TRIVIAL(info) << "At time " << scheduler->getSimTime()
618  << " user " << dest.first << "," << dest.second
619  << " started waiting for chunk " << completedChunk+1 << " of content "
620  << flow->getContent()->getName();
621  BOOST_LOG_TRIVIAL(info) << "Highest chunk fetched so far: "
622  << userWatchMap.at(dest).highestChunkFetched;
623  userWatchMap.at(dest).waiting = true;
624  }
625  }
626  break;
627  default:
628  BOOST_LOG_TRIVIAL(error) << "notifyCompletedFlow for non TRANSFER, non WATCH event.";
629  abort();
630  }

825 std::pair<bool, bool> TopologyOracle::optimizeCaching(PonUser reqUser,
826  const ChunkPtr& chunk, SimTime time, uint currentRound) {
827  SimTime absTime = time + (currentRound*roundDuration);
828  /* This should no longer apply as chunks are either cached entirely or not cached
829  *
830  * there is an inconsistence in our assumptions here, in that we add
831  * an optimization variable for the requested content in addition to the
832  * variables we have for the elements cached, but there is a chance that the
833  * requested content was already cached (albeit with a different size). I need
834  * to either make sure that this case is covered or that it never happens (e.g.,
835  * by deleting it). In the next few lines I attempt to do so.
836 
837  if (checkIfCached(reqUser, content) == true) {
838  userCacheMap->at(reqUser).removeFromCache(content, absTime);
839  removeFromCMap(content, reqUser);
840  }
841  */
842  int hour = std::floor(time / 3600);
843  /* attempting to fix a potential bug in case we enter this function when it's
844  /* midnight of the following day
845  */
846  if (hour >= usrPctgByHour.size())
847  hour = usrPctgByHour.size() - 1;
848  IloEnv env;
849  int asid = topo->getAsid(reqUser);
850  try {
851  IloModel model(env);
852  IloNumVarArray c(env);
853  auto cachedVec = userCacheMap->at(reqUser).getCacheMap();
854  // add a boolean variable for each of the elements cached
855  // char varName[24];
856  for (auto it = cachedVec.begin(); it != cachedVec.end(); it++) {
857  // varName = ((string)("c_" + it->first->getName())).c_str();
858  c.add(IloNumVar(env, 0, 1, ILOINT));
859  }
860  // add a boolean variable for the element requested
861  // varName = ((string)("c_" + content->getName())).c_str();
862  c.add(IloNumVar(env, 0, 1, ILOINT));
863  // Define the constraints
864  IloRangeArray constraints(env);
865  // Initialize the objective function expression
866  IloNumExpr exp(env);
867  // add the size of the elements cached, if kept in the cache
868  int i = 0;
869  for (auto it = cachedVec.begin(); it != cachedVec.end(); it++) {
870  ChunkPtr chunkIt = it->first;
871  ContentElement* contIt = chunkIt->getContent();
872  exp += c[i] * it->second.size;
873  // also make sure that we do not erase a content if we are uploading it
874  if (it->second.uploads > 0) {
875  constraints.add(c[i] == 1);
876  i++;
877  continue;
878  }
879  // build the constraint on rates if required
880  /* the contentRateVec holds the number of requests expected for each content
881  * per day. To get the number of requests per hour per AS we need
882  * to get the % of requests in this particular hour via usrPctgByHour and
883  * multiply it by the fraction of users in the AS of the requester. The avg
884  * concurrent request number is calculated as reqPerHour * avgReqLength. To
885  * get the peak of concurrent users, which is what we need, we multiply
886  * that value by a pakReqRatio, which is taken as an integer input parameter
887  * (-k).
888  */
889  uint dayIndex = currentRound - contIt->getReleaseDay();
890  assert(0 <= dayIndex && dayIndex < 7);
891  /* if it's a young content (released today and with only a few hours of life)
892  * make sure it is not erased, as ranking is still too volatile to be accurate
893  */
894  if (dayIndex == 0 && hour < 6) {
895  constraints.add(c[i] == 1);
896  i++;
897  continue;
898  }
899  uint rank = dailyRanking.at(dayIndex).getRankOf(contIt);
900  double rate = contentRateVec.at(dayIndex).at(rank);
901  double avgReqPerHour = (rate * usrPctgByHour.at(hour) / 100) *
902  (topo->getASCustomers(asid) / topo->getNumCustomers());
903  BOOST_LOG_TRIVIAL(trace) << "avgReqPerHour(" << contIt->getName() <<
904  ":" << chunkIt->getIndex() << "," << hour
905  << ") = (" << rate << " * " << usrPctgByHour.at(hour)
906  << " / 100) * (" << topo->getASCustomers(asid) << " / " << topo->getNumCustomers()
907  << ") = " << avgReqPerHour;
908  if (avgReqPerHour < 1) {
909  BOOST_LOG_TRIVIAL(trace) << "avgReqPerHour for chunk " << chunkIt->getIndex()
910  << " of content " << contIt->getName()
911  << " is less than 1 (" << avgReqPerHour << "), setting it to 1";
912  avgReqPerHour = 1;
913  }
914  int chunkRate = std::floor((peakReqRatio * avgReqPerHour * (avgReqLength / 3600)) + 0.5);
915  // ensure that the peakReqRatio does not inflate the requests to more than the users we have
916  chunkRate = std::min(chunkRate, static_cast<int>(topo->getASCustomers(asid)));
917  // ensure that we keep at least one copy of each content in each AS
918  chunkRate = std::max(1, chunkRate);
919  BOOST_LOG_TRIVIAL(trace) << "chunkRate = std::floor((" << peakReqRatio
920  << " * " << avgReqPerHour
921  << " * " << avgReqLength << " / 3600) + 0.5) = " << chunkRate;
922  IloNumExpr cExp(env, 0);
923  for (auto uit = asidContentMap->at(asid).at(chunkIt).begin();
924  uit != asidContentMap->at(asid).at(chunkIt).end(); uit++) {
925  if (reqUser != *uit) {
926  cExp += maxUploads - userCacheMap->at(*uit).getTotalUploads()
927  + userCacheMap->at(*uit).getCurrentUploads(chunkIt);
928  } else {
929  // multiply the term by the caching variable so that we lose it if
930  // we decide to erase the element from the cache
931  cExp += c[i]* (IloInt) (maxUploads - userCacheMap->at(reqUser).getTotalUploads()
932  + userCacheMap->at(reqUser).getCurrentUploads(chunkIt));
933  }
934  }
935  constraints.add(cExp >= chunkRate);
936  i++;
937  }
938  // same for the chunk we just downloaded
939  exp += c[cachedVec.size()] * chunk->getSize();
940  // add a chunkRate constraint for the downloaded chunk if needed
941  uint dayIndex = currentRound - chunk->getContent()->getReleaseDay();
942  assert(0 <= dayIndex && dayIndex < 7);
943  if (dayIndex == 0 && hour < 6) {
944  constraints.add(c[cachedVec.size()] == 1);
945  } else {
946  uint rank = dailyRanking.at(dayIndex).getRankOf(chunk->getContent());
947  double rate = contentRateVec.at(dayIndex).at(rank);
948  double avgReqPerHour = (rate * usrPctgByHour.at(hour) / 100) *
949  (topo->getASCustomers(asid) / topo->getNumCustomers());
950  BOOST_LOG_TRIVIAL(trace) << "avgReqPerHour(" << chunk->getContent()->getName()
951  << ":" << chunk->getIndex() << "," << hour
952  << ") = (" << rate << " * " << usrPctgByHour.at(hour)
953  << " / 100) * (" << topo->getASCustomers(asid) << " / " << topo->getNumCustomers()
954  << ") = " << avgReqPerHour;
955  if (avgReqPerHour < 1) {
956  BOOST_LOG_TRIVIAL(trace) << "avgReqPerHour for chunk " << chunk->getIndex()
957  << " of content " << chunk->getContent()->getName()
958  << " is less than 1 (" << avgReqPerHour << "), setting it to 1";
959  avgReqPerHour = 1;
960  }
961  int chunkRate = std::floor((peakReqRatio * avgReqPerHour * (avgReqLength / 3600)) + 0.5);
962  // ensure that the peakReqRatio does not inflate the requests to more than the users we have
963  chunkRate = std::min(chunkRate, static_cast<int>(topo->getASCustomers(asid)));
964  // ensure that we keep at least one copy of each content in each AS
965  chunkRate = std::max(1, chunkRate);
966  BOOST_LOG_TRIVIAL(trace) << "chunkRate = std::floor((" << peakReqRatio
967  << " * " << avgReqPerHour
968  << " * " << avgReqLength << " / 3600) + 0.5) = " << chunkRate;
969  IloNumExpr cExp(env, 0);
970  for (auto uit = asidContentMap->at(asid).at(chunk).begin();
971  uit != asidContentMap->at(asid).at(chunk).end(); uit++) {
972  if (reqUser != *uit) {
973  cExp += maxUploads - userCacheMap->at(*uit).getTotalUploads()
974  + userCacheMap->at(*uit).getCurrentUploads(chunk);
975  } else {
976  cExp += c[cachedVec.size()]* (IloInt) (maxUploads - userCacheMap->at(reqUser).getTotalUploads()
977  + userCacheMap->at(reqUser).getCurrentUploads(chunk));
978  }
979  }
980  // if the requesting user was not already caching the content, we must add
981  // its term to the sum in case we decide to cache the requested content
982  if (userCacheMap->at(reqUser).isCached(chunk) == false) {
983  cExp += c[cachedVec.size()]* (IloInt) (maxUploads - userCacheMap->at(reqUser).getTotalUploads()
984  + userCacheMap->at(reqUser).getCurrentUploads(chunk));
985  }
986  constraints.add(cExp >= chunkRate);
987  }
988  //finally add a constraint on the maximum size of the cache
989  constraints.add(exp <= userCacheMap->at(reqUser).getMaxSize());
990 
991  // specify that we are interested in minimizing the objective
992  IloObjective obj = IloMinimize(env, exp);
993  // Add the objectives and the constraints to the model
994  model.add(c);
995  model.add(obj);
996  model.add(constraints);
997 
998  // ask CPLEX to solve the problem
999  IloCplex cplex(model);
