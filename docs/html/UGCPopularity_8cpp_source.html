26 // generate views for all the content elements at a given phase for this round only
27 unsigned int UGCPopularity::generateViews(std::list<ContentElement*> contentList,
28  peakingPhase phase) {
29  unsigned int bodyCount, tailCount;
30  tailCount = std::floor(contentList.size() / 10);
31  bodyCount = contentList.size() - tailCount;
32  std::list<unsigned int> views;
33  if (phase == BEFORE_PEAK) {
34  for (unsigned int i = 0; i < tailCount; i++) {
35  views.push_back(this->getBeforePeakTailViews());
36  }
37  for (unsigned int i = 0; i < bodyCount; i++) {
38  views.push_back(this->getBeforePeakViews());
39  }
40  } else if (phase == AT_PEAK) {
41  for (unsigned int i = 0; i < tailCount; i++) {
42  views.push_back(this->getAtPeakTailViews());
43  }
44  for (unsigned int i = 0; i < bodyCount; i++) {
45  views.push_back(this->getAtPeakViews());
46  }
47  } else { // phase == AFTER_PEAK
48  for (unsigned int i = 0; i < tailCount; i++) {
49  views.push_back(this->getAfterPeakTailViews());
50  }
51  for (unsigned int i = 0; i < bodyCount; i++) {
52  views.push_back(this->getAfterPeakViews());
53  }
54  }
55  // sort the views in descending order and assign them to the contentList
56  // (which is ordered by ranking) so that relative ranking remains unaltered
57  views.sort(std::greater<unsigned int>());
58  std::list<unsigned int>::iterator viewsIt, mIt;
59 
60  // introduce perturbations in the number of views, by switching views between
61  // videos in the same phase and at a maximum distance g
62  if (this->perturbations && phase != AT_PEAK) {
63  uint g = 12; // suggested value from Borghol et al.
64  std::vector<std::pair<uint, uint> > winLimits;
65  uint minWin, maxWin;
66  uint maxViews = views.front();
67  // compute valid switching windows for each content
68  BOOST_FOREACH(uint v, views) {
69  minWin = std::floor(v / g);
70  maxWin = std::min(maxViews, v * g);
71  winLimits.push_back(std::make_pair(minWin,maxWin));
72  }
73  // switch views a "sufficiently large number of times" (!) between contents
74  // with compatible switching windows
75 
76  // NOTE: the following code is so ugly it makes me want to kill kittens.
77  // try to rewrite it in a half-decent way, ok?
78  boost::random::uniform_int_distribution<> uniDist(0, contentList.size()-1);
79  for (uint i = 0; i < contentList.size() / 3; i++) {
80  uint randC = uniDist(gen);
81  viewsIt = views.begin();
82  for (uint j = 0; j < randC; j++)
83  viewsIt++;
84  mIt = viewsIt;
85  int j = 0;
86  while (mIt != views.begin() && *mIt >= winLimits[randC].first) {
87  j--;
88  mIt--;
89  }
90  if (j != 0) j++;
91  mIt = viewsIt;
92  int k = 0;
93  while (mIt != views.end() && *mIt <= winLimits[randC].second) {
94  mIt++;
95  k++;
96  }
97  if (k != 0) k--;
98  if (j != 0 || k != 0) {
99  boost::random::uniform_int_distribution<> switchDist(0,k-j);
100  j = switchDist(gen) + j;
101  if (j < 0) {
102  std::swap(winLimits[randC], winLimits[randC+j]);
103  mIt = viewsIt;
104  while (j < 0) {
105  mIt--;
106  j++;
107  }
108  std::iter_swap(viewsIt, mIt);
109  }
110  else if (j > 0) {
111  std::swap(winLimits[randC], winLimits[randC+j]);
112  mIt = viewsIt;
113  while (j > 0) {
114  mIt++;
115  j--;
116  }
117  std::iter_swap(viewsIt, mIt);
118  }
119  }
120  }
121  }
122  // assign views in the intended order (sorted or permuted)
123  uint totalViews = 0;
124  viewsIt = views.begin();
125  BOOST_FOREACH (ContentElement* content, contentList) {
126  content->setViewsThisRound(*viewsIt);
127  totalViews += *viewsIt;
128  viewsIt++;
129  }
130  return totalViews;
131 }
