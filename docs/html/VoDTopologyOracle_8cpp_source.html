105 void VoDTopologyOracle::updateCatalog(uint currentRound) {
106  // move content that peaked last round to afterPeak
107  afterPeak.splice(afterPeak.begin(), atPeak);
108  // move content that peaks in the next round from beforePeak to atPeak
109  std::list<ContentElement*>::iterator temp;
110  std::list<ContentElement*>::iterator it = beforePeak.begin();
111  while (it != beforePeak.end()) {
112  if ((*it)->getPeakingRound() == currentRound + 1) {
113  temp = it;
114  temp++;
115  atPeak.splice(atPeak.end(), beforePeak, it);
116  it = temp;
117  } else
118  it++;
119  }
