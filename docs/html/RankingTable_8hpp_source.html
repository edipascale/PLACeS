70 void RankingTable<Element>::insert(Element e) {
71  // check if the element is already in the catalog
72  if (catalog.left.find(e) != catalog.left.end())
73  return;
74  // insert it at the end
75 
76  catalog.left.insert(std::make_pair(e,hits.size()));
77  this->hits.push_back(0);
78  this->roundHits.push_back(0);
79  assert(hits.size() == catalog.left.size());
80  assert(hits.size() == roundHits.size());

102 template <class Element>
103 void RankingTable<Element>::eraseByRank(uint rank) {
104  auto init = catalog.right.find(rank);
105  if (init != catalog.right.end()) {
106  catalog.right.erase(init);
107  hits.erase(hits.begin()+rank);
108  roundHits.erase(roundHits.begin()+rank);
109  assert(catalog.right.size() == hits.size());
110  assert(hits.size() == roundHits.size());
111  init = catalog.right.find(rank+1);
112  for (auto it = init; it != catalog.right.end(); it++) {
113  catalog.right.replace_key(it, rank);
114  rank++;
115  }
116  }
117 }

219 void RankingTable<Element>::hit(Element e) {
220  auto it = catalog.left.find(e);
221  if (it == catalog.left.end())
222  throw -1;
223  uint oldRank = it->second;
224  uint oldHits = hits.at(oldRank);
225  hits.at(oldRank) = hits.at(oldRank) + 1;
226  roundHits.at(oldRank) = roundHits.at(oldRank) + 1;
227  int index = 0;
228  for (index = oldRank; index > 0 && hits.at(index-1) < hits.at(oldRank); index--);
229  if (index != oldRank) {
230 // std::cout << "debug: swapping " << oldRank << "," << index << std::endl;
231  bool success = swap(oldRank, index);
232 // printRanking();
233  assert(success);
