188 void Scheduler::updateSchedule(Flow* flow, SimTime oldEta) {
189  if (handleMap.find(flow) != handleMap.end()) {
190  if (flow->getEta() > oldEta)
191  pendingEvents.decrease(handleMap[flow]);
192  else if (flow->getEta() < oldEta)
193  pendingEvents.increase(handleMap[flow]);

196  }

204 void Scheduler::startNewRound() {
205  // update flows so that they can be moved to the new round
206  std::vector<Flow*> flowVec;
207  while (pendingEvents.empty() == false) {
208  Flow* f = const_cast<Flow*> (pendingEvents.top());
209  pendingEvents.pop();
210  if (f->getFlowType() == FlowType::WATCH) {
211  // there is no point in carrying over watch flows, as they will be discarded
212  BOOST_LOG_TRIVIAL(trace) << "Deleting watch flow for chunk" << f->getChunkId()
213  << " of content " << f->getContent()->getName() << " for user "
214  << f->getDestination().first << "," << f->getDestination().second;
215  delete f;
216  continue;
217  }
218  f->updateSizeDownloaded(this->roundDuration);
219  f->setLastUpdate(0);
220  f->setStart(f->getStart() - this->roundDuration); // negative
221  SimTime oldEta = f->getEta();
222  if (oldEta < this->roundDuration) {
223  BOOST_LOG_TRIVIAL(error) << "Scheduler::startNewRound() - unresolved event has eta "
224  << oldEta << " < roundDuration";
225  abort();
226  }
227  f->setEta(oldEta - this->roundDuration);
228  if (this->mode == IPTV && f->getContent()->getReleaseDay() <= currentRound-6) {
229  // expired contents in IPTV will be deleted so flows can't be completed,
230  // this event should have been truncated last round
231  BOOST_LOG_TRIVIAL(info) << "Scheduler::startNewRound() - carried over flow "
232  "with expired content will not be completed";
233  if (f->getFlowType() == FlowType::TRANSFER) {
234  // this is hacky, but needs to be done. Ideally we should not get here at all.
235  oracle->getTopology()->updateCapacity(f, this, false);
236  f->setContent(nullptr);
237  }
238  } else
239  flowVec.push_back(f);
240  }
241  handleMap.clear();
242  BOOST_FOREACH (Flow* f, flowVec) {
243  this->schedule(f);
244  }
245  flowVec.clear();
246  this->simTime = 0;
247  this->currentRound++;
248  this->terminate = new Flow(nullptr, UNKNOWN, roundDuration);
249  this->terminate->setFlowType(FlowType::TERMINATE);
250  this->schedule(terminate);
251  if (snapshotFreq > 0 && snapshotFreq <= roundDuration) {
252  this->snapshot = new Flow(nullptr, UNKNOWN, snapshotFreq);
253  snapshot->setFlowType(FlowType::SNAPSHOT);
254  this->schedule(this->snapshot);
255  } else {
256  this->snapshot = nullptr;
257  }
258  return;
259 }
