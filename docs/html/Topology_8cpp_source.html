345 std::vector<Edge> Topology::getRoute(uint source, uint dest) {
346  VertexVec visitedNodes;
347  std::vector<Edge> route;
348  visitedNodes.clear();
349  Vertex currentNode = boost::vertex(dest, topology);
350  Vertex sourceV = boost::vertex(source, topology);
351  if (topology[source].ponCustomers > 0) {
352  // as ONU nodes have no pMap entry, we need to go to the metro/core node and
353  // route from there. Incidentally, this also solves the issue with users
354  // on the same PON exchanging data (the route goes to the m/c node as it
355  // should)
356  // NOTE: as the graph is not bidirectedS, in_edges is not available
357  assert(boost::out_degree(source, topology) == 1);
358  Edge outEdge = *(boost::out_edges(source, topology).first);
359  sourceV = boost::target(outEdge, topology);
360  }
361 
362  while (currentNode != sourceV) {
363  visitedNodes.push_back(currentNode);
364  currentNode = pMap[sourceV][currentNode];
365  }
366  visitedNodes.push_back(sourceV);
367  if (topology[source].ponCustomers > 0)
368  visitedNodes.push_back(source);
369  // Walk back through the visited nodes and record the edges in the right order
370  std::pair<Edge, bool> returnValue;
371  for (VertexVec::reverse_iterator rvIt = visitedNodes.rbegin();
372  rvIt+1 != visitedNodes.rend(); rvIt++) {
373  returnValue = boost::edge(*rvIt, *(rvIt+1), topology);
374  // assert(returnValue.second == true);
375  if (!returnValue.second) {
376  std::cerr << "Topology::getRoute() - Failed to retrieve edge from Vertex "
377  << *(rvIt-1) << " to Vertex " << *rvIt << std::endl;
378  exit(ERR_FAILED_ROUTING);
379  }
380  route.push_back(returnValue.first);
381  }
382  return route;
383 }

389 void Topology::updateCapacity(Flow* flow, Scheduler* scheduler,
390  bool addNotRemove) {
391  if (flow->getFlowType() != FlowType::TRANSFER) {
392  BOOST_LOG_TRIVIAL(warning) << "Topology::updateCapacity() called for a non TRANSFER flow";
393  return;
394  }
395  // if the chunk is smaller than MAX_FLOW_SPEED, that is the maximum bandwidth
396  // we should be able to get in Mbps!
397  Capacity MaxBwAchievable = std::min(flow->getChunkSize(), MAX_FLOW_SPEED);
398  SimTime now = scheduler->getSimTime();
399  std::vector<Edge> flowRoute = this->getRoute(flow->getSource(),
400  flow->getDestination());
401  Capacity minSpareCapacity(UNLIMITED), minCut(UNLIMITED), maxBneckBw(UNLIMITED);
402  Edge bottleneck = flowRoute.back();
403  if (addNotRemove) {
404  // new flow, traverse route to determine minSpareCapacity and minCut
405  BOOST_FOREACH (Edge e, flowRoute) {
406  topology[e].activeFlows.insert(flow);
407  minSpareCapacity = std::min(minSpareCapacity, topology[e].spareCapacity);
408  if (topology[e].maxCapacity != UNLIMITED &&
409  topology[e].maxCapacity / topology[e].activeFlows.size() < minCut) {
410  minCut = topology[e].maxCapacity / topology[e].activeFlows.size();
411  bottleneck = e;
412  }
413  }
414  if (minSpareCapacity >= MaxBwAchievable) {
415  // there's enough spare capacity, no need to reduce other flows' bw
416  flow->setBandwidth(MaxBwAchievable);
417  this->updateRouteCapacity(flowRoute, -MaxBwAchievable);
418  this->updateEta(flow, scheduler);
419  return;
420  } else {
421  // assign bw basing on the bottleneck for this flow
422  // rounding to prevent capacity overflow, but need to ensure that we
423  // don't round to 0!
424  if (std::floor(minCut) > 0)
425  minCut = std::floor(minCut);
426  assert(minCut > 0);
427  flow->setBandwidth(minCut);
428  this->updateRouteCapacity(flowRoute, -minCut);
429  this->updateEta(flow, scheduler);
430  // reduce bw to other flows to match link capacity constraints
431  BOOST_FOREACH(Flow* f, topology[bottleneck].activeFlows) {
432  if (f->getBandwidth() > minCut) {
433  this->updateRouteCapacity(this->getRoute(f->getSource(), f->getDestination()),
434  f->getBandwidth() - minCut);
435  f->updateSizeDownloaded(now);
436  f->setBandwidth(minCut);
437  this->updateEta(f, scheduler);
438  }
439  }
440  // we need this check to account for rounding errors.
441  if (topology[bottleneck].spareCapacity < 0) {
442  // std::cerr << "WARNING: Topology::updateCapacity() - negative spareCapacity ("
443  // << topology[bottleneck].spareCapacity << std::endl;
444  topology[bottleneck].spareCapacity = 0;
445  }
446  }
447  }
448  else {
449  // flow removal is more tricky, as you really can't be conservative when
450  // increasing capacity to a flow - you have to check that you've got
451  // enough spare capacity on other links.
452  // free bw across the route and check how much bw are other flows consuming
453  BOOST_FOREACH(Edge e, flowRoute) {
454  topology[e].activeFlows.erase(flow);
455  // if this link has infinite capacity, we're done here
456  if (topology[e].maxCapacity != UNLIMITED) {
457  topology[e].spareCapacity += flow->getBandwidth();
458  // minCut is the minimum (across links) of the average bw of impacted flows
459  if (topology[e].activeFlows.size() > 0)
460  minCut = std::min(minCut, (topology[e].maxCapacity - topology[e].spareCapacity)
461  / topology[e].activeFlows.size());
462  // maxBneckBw is the maximum bw a flow can get with fair sharing on the
463  // bottleneck for the removed flow route
464  if (topology[e].activeFlows.size() > 0 &&
465  topology[e].maxCapacity / topology[e].activeFlows.size() < maxBneckBw) {
466  maxBneckBw = topology[e].maxCapacity / topology[e].activeFlows.size();
467  bottleneck = e;
468  }
469  }
470  }
471  if (minCut >= MAX_FLOW_SPEED) {
472  // every other flow is already at max speed, no need to do anything else
473  return;
474  }
475  else {
476  // rounding to prevent capacity overflow, but need to ensure that we
477  // don't round to 0!
478  if (std::floor(maxBneckBw) > 0)
479  maxBneckBw = std::floor(maxBneckBw);
480  maxBneckBw = std::min (MAX_FLOW_SPEED, maxBneckBw);
481  std::vector<Edge> fRoute;
482  BOOST_FOREACH (Flow* f, topology[bottleneck].activeFlows) {
483  // check if there's room to increase this flow bw (basing only on the
484  // removed flow bottleneck, hence not optimal as the previous method)
485  // remember that we can only get the chunk's size worth of bw
486  MaxBwAchievable = std::min(f->getChunkSize(), maxBneckBw);
487  Capacity increase = MaxBwAchievable - f->getBandwidth();
488  if (increase > this->minFlowIncrease) {
489  fRoute = this->getRoute(f->getSource(), f->getDestination());
490  bool safeToGrow = true;
491  BOOST_FOREACH (Edge e, fRoute) {
492  if (topology[e].spareCapacity <= increase ) {
493  safeToGrow = false;
494  break;
495  }
496  }
497  if (safeToGrow) {
498  // we've got enough capacity to increase this flow's bw
499  this->updateRouteCapacity(fRoute, -increase);
500  f->updateSizeDownloaded(now);
501  f->setBandwidth(MaxBwAchievable);
502  // this if clause was added to prevent a failed assertion due to flows carried over from previous rounds
503  if (now <= scheduler->getRoundDuration())
504  this->updateEta(f, scheduler);
505  }
506  fRoute.clear();
507  }
508  }
509  // we need this check to account for rounding errors.
510  if (topology[bottleneck].spareCapacity < 0) {
511  // std::cerr << "WARNING: Topology::updateCapacity() - negative spareCapacity ("
512  // << topology[bottleneck].spareCapacity << std::endl;
513  topology[bottleneck].spareCapacity = 0;
514  }
515  }
516  }

523 void Topology::printNetworkStats(uint currentRound, uint roundDuration) {
524  Capacity averageTot(0), averageCore(0), averageAccessUp(0), temp(0), max(0);
525  Capacity peakAccessUp(0), peakCore(0), avgPeakAccessUp(0), avgPeakCore(0);
526  Capacity averageAccessDown(0), avgPeakAccessDown(0), peakAccessDown(0), tempPeak(0);
527  Capacity averageMetro(0), peakMetro(0), avgPeakMetro(0);
528  /* Take the first edge of the graph and use it to initialize the following
529  * variables; this is required because, if all the core edges have unlimited
530  * capacity, peakCoreEdge would otherwise be returned as an undefined variable
531  * in the previous implementation.
532  */
533  Edge nullE = *(boost::edges(topology).first) ;
534  Edge maxEdge(nullE), peakAccessUpEdge(nullE), peakAccessDownEdge(nullE),
535  peakCoreEdge(nullE), peakMetroEdge(nullE);
536  uint numAccessEdges = (numEdges - (numCoreEdges + numMetroEdges)) / 2;
537  BOOST_FOREACH (Edge e, boost::edges(topology)) {
538  if (loadMap[e] > 0) {
539  temp = loadMap[e] / roundDuration;
540  tempPeak = topology[e].peakCapacity;

542  if (this->getEdgeType(e) == CORE) {
543  averageCore += temp;
544  avgPeakCore += tempPeak;
545  if (tempPeak > peakCore) {
546  peakCore = tempPeak;
547  peakCoreEdge = e;
548  }
549  }
550  else if (this->getEdgeType(e) == METRO) {
551  averageMetro += temp;
552  avgPeakMetro += tempPeak;
553  if (tempPeak > peakMetro) {
554  peakMetro = tempPeak;
555  peakMetroEdge = e;
556  }
557  }
558  else if (this->getEdgeType(e) == UPSTREAM) {
559  averageAccessUp += temp;
560  avgPeakAccessUp += tempPeak;
561  if (tempPeak > peakAccessUp) {
562  peakAccessUp = tempPeak;
563  peakAccessUpEdge = e;
564  }
565  }
566  else if (this->getEdgeType(e) == DOWNSTREAM) {
567  averageAccessDown += temp;
568  avgPeakAccessDown += tempPeak;
569  if (tempPeak > peakAccessDown) {
570  peakAccessDown = tempPeak;
571  peakAccessDownEdge = e;
572  }
573  }
574  if (temp > max) {
575  max = temp;
576  maxEdge = e;
577  }
578  }
579  }
580  // store the values in the stats vectors for future access or manipulation
581  stats.avgTot.at(currentRound) = (Capacity)(averageTot / numEdges);
582  stats.avgCore.at(currentRound) = (Capacity)(averageCore / numCoreEdges);
583  stats.avgAccessUp.at(currentRound) = (Capacity)(averageAccessUp / numAccessEdges);
584  stats.avgAccessDown.at(currentRound) = (Capacity)(averageAccessDown / numAccessEdges);
585  stats.peakCore.at(currentRound) = peakCore;
586  stats.peakAccessUp.at(currentRound) = peakAccessUp;
587  stats.peakAccessDown.at(currentRound) = peakAccessDown;
588  stats.avgPeakCore.at(currentRound) = (Capacity)(avgPeakCore / numCoreEdges);
589  stats.avgPeakAccessUp.at(currentRound) = (Capacity)(avgPeakAccessUp / numAccessEdges);
590  stats.avgPeakAccessDown.at(currentRound) = (Capacity)(avgPeakAccessDown / numAccessEdges);
591  if (numMetroEdges == 0) {
592  stats.avgMetro.at(currentRound) = 0;
593  stats.avgPeakMetro.at(currentRound) = 0;
594  stats.peakMetro.at(currentRound) = 0;
595  }
596  else {
597  stats.avgMetro.at(currentRound) = (Capacity)(averageMetro / numMetroEdges);
598  stats.avgPeakMetro.at(currentRound) = (Capacity)(avgPeakMetro / numMetroEdges);
599  stats.peakMetro.at(currentRound) = peakMetro;
600  }
601  // Print stats on the screen for human visualization
602  std::cout << "Average load: "
603  << stats.avgTot.at(currentRound)
604  << " (core: " << stats.avgCore.at(currentRound)
605  << "; metro: " << stats.avgMetro.at(currentRound)
606  << "; access_up: " << stats.avgAccessUp.at(currentRound)
607  << "; access_down: " << stats.avgAccessDown.at(currentRound)
608  << "); maximum average load on edge " << maxEdge.m_source
609  << "-" << maxEdge.m_target << " ("
610  << max << ")" << std::endl;
611  std::cout << "Average peak core load: "
612  << stats.avgPeakCore.at(currentRound)
613  << ", maximum peak core load on edge " << peakCoreEdge.m_source
614  << "-" << peakCoreEdge.m_target << " ("
615  << stats.peakCore.at(currentRound) << ")" << std::endl;
616  std::cout << "Average peak metro load: "

621  std::cout << "Average peak upstream access load: "

624  << "-" << peakAccessUpEdge.m_target << " ("

696 void Topology::updateRouteCapacity(std::vector<Edge> route, Capacity toAdd) {
697  BOOST_FOREACH(Edge e, route) {

705  topology[e].maxCapacity);

707  topology[e].peakCapacity = usedBw;

709  }

713 void Topology::updateEta(Flow* flow, Scheduler* scheduler) {
714  /* This is only called by Topology::updateCapacity() and only applies to
715  * FlowType::TRANSFER flows, no need to worry about zapping.
716  */
717  SimTime downloadEta, oldEta(flow->getEta());
718  SimTime now = scheduler->getSimTime();
719  downloadEta = now + std::floor(((flow->getChunkSize() - flow->getSizeDownloaded())
720  / flow->getBandwidth()) + 0.5);
721  flow->setEta(downloadEta);
722  // Ensure that at least 1 second of flow is simulated
723  if (flow->getEta() <= flow->getStart())
724  flow->setEta(flow->getStart() + 1);
725  assert(flow->getEta() >= now);
726  if (flow->getEta() != oldEta) {
727  scheduler->updateSchedule(flow, oldEta);
728  BOOST_LOG_TRIVIAL(debug) << " At time " << now << " updated ETA of flow "
729  << flow->toString() << " from " << oldEta << " to " << flow->getEta();
730  }
731 }
